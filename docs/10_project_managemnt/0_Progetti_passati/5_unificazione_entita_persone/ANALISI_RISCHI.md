// Alert system
const checkRisks = async () => {
  for (const [riskId, config] of Object.entries(riskMonitoring)) {
    const currentValue = await config.check();

        risk: riskId,
        metric: config.metric,
        current: currentValue,
        threshold: config.threshold,
